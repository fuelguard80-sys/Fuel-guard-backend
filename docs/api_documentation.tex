\section{Authentication  \normalfont\small\texttt{/api/v1/auth}}
% ══════════════════════════════════════════════════════════════════════════════

\begin{endpoint}{\POST}{/auth/signup}{\noauth}{
  Registers a new user, creates their Firestore profile and mirrors the account into
  Firebase Auth so email-verification and password-reset emails can be sent.
}
  \reqheader
  \paramtable{
    email        & string  & Valid email address \\
    password     & string  & Minimum 8 characters \\
    full\_name   & string  & Minimum 2 characters \\
    phone        & string? & Optional; Pakistani format \\
    role         & enum    & \texttt{customer | employee | admin | driver | super\_admin} \\
  }
  \resheader
  \paramtable{
    access\_token  & string & JWT; expires in 60 min \\
    refresh\_token & string & JWT; expires in 30 days \\
    token\_type    & string & Always \texttt{"bearer"} \\
    uid            & string & Firestore user document ID \\
    role           & string & Assigned role \\
  }
\end{endpoint}

\begin{endpoint}{\POST}{/auth/login}{\noauth}{
  Authenticates a user with email and password. If the bcrypt hash is stale
  (e.g.\ after a browser-based Firebase password reset), falls back to Firebase
  Auth and self-heals the Firestore hash.
}
  \reqheader
  \paramtable{
    email    & string & Registered email \\
    password & string & Account password \\
  }
  \resheader
  \paramtable{
    access\_token  & string & JWT access token \\
    refresh\_token & string & JWT refresh token \\
    token\_type    & string & \texttt{"bearer"} \\
    uid            & string & User ID \\
    role           & string & User role \\
  }
\end{endpoint}

\begin{endpoint}{\POST}{/auth/logout}{\authrequired}{
  Stateless logout — the client discards its tokens. No server-side token
  revocation is performed (add a Firestore denylist for full revocation).
}
  \reqheader\quad\textit{None (Auth header only)}\\[4pt]
  \resheader
  \paramtable{
    message & string & \texttt{"Logged out successfully"} \\
  }
\end{endpoint}

\begin{endpoint}{\POST}{/auth/refresh-token}{\noauth}{
  Exchanges a valid refresh token for a new access/refresh token pair without
  requiring the user to log in again.
}
  \reqheader
  \paramtable{
    refresh\_token & string & Valid, non-expired refresh JWT \\
  }
  \resheader
  \paramtable{
    access\_token  & string & New JWT access token \\
    refresh\_token & string & New JWT refresh token \\
    uid            & string & User ID \\
    role           & string & User role \\
  }
\end{endpoint}

\begin{endpoint}{\POST}{/auth/forgot-password}{\noauth}{
  Triggers a Firebase Auth password-reset email to the given address. Always
  returns the same message regardless of whether the email is registered,
  preventing user-enumeration attacks.
}
  \reqheader
  \paramtable{
    email & string & Email to send the reset link to \\
  }
  \resheader
  \paramtable{
    message & string & Generic confirmation message \\
  }
\end{endpoint}

\begin{endpoint}{\POST}{/auth/reset-password}{\noauth}{
  Syncs a Firebase-confirmed password reset back to Firestore by verifying the
  Firebase ID token and updating the bcrypt hash, then returns fresh JWT tokens
  so the user is immediately logged in.
}
  \reqheader
  \paramtable{
    firebase\_id\_token & string & Firebase ID token from \texttt{signInWithEmailAndPassword} \\
    new\_password       & string & New password, minimum 8 characters \\
  }
  \resheader
  \paramtable{
    message        & string & Confirmation \\
    access\_token  & string & JWT access token \\
    refresh\_token & string & JWT refresh token \\
    uid            & string & User ID \\
    role           & string & User role \\
  }
\end{endpoint}

\begin{endpoint}{\POST}{/auth/chatbot}{\authrequired}{
  Sends a user message to the FuelGuard AI support assistant powered by
  Groq \texttt{llama-3.3-70b-versatile}; responds only to FuelGuard-related queries.
}
  \reqheader
  \paramtable{
    message & string & 1–1000 characters \\
  }
  \resheader
  \paramtable{
    reply & string & AI-generated response \\
  }
\end{endpoint}

\section{Users  \normalfont\small\texttt{/api/v1/users}}
% ══════════════════════════════════════════════════════════════════════════════

\begin{endpoint}{\GET}{/users/me}{\authrequired}{
  Returns the full profile of the currently authenticated user, including
  verification status and avatar URL.
}
  \reqheader\quad\textit{None}\\[4pt]
  \resheader
  \paramtable{
    uid            & string  & Unique user ID \\
    email          & string  & Email address \\
    full\_name     & string  & Display name \\
    phone          & string? & Phone number \\
    role           & enum    & User role \\
    avatar\_url    & string? & Cloudinary URL \\
    is\_active     & bool    & Account status \\
    email\_verified & bool   & Firebase email verification status \\
    created\_at    & datetime & Registration timestamp \\
  }
\end{endpoint}

\begin{endpoint}{\PUT}{/users/me}{\authrequired}{
  Updates the authenticated user's editable profile fields. Only provided
  fields are updated; omitted fields remain unchanged.
}
  \reqheader
  \paramtable{
    full\_name  & string? & New display name \\
    phone       & string? & New phone number \\
    avatar\_url & string? & Direct URL (use \texttt{POST /users/me/avatar} for uploads) \\
  }
  \resheader\quad\textit{Full \texttt{UserProfile} object (same as GET /users/me)}
\end{endpoint}

\begin{endpoint}{\POST}{/users/me/avatar}{\authrequired}{
  Uploads a profile picture as a multipart file, stores it on Cloudinary under
  \texttt{fuelguard/avatars/\{uid\}/} with a 512×512 face-crop transform, and saves
  the returned URL to Firestore.
}
  \reqheader
  \paramtable{
    file & image (multipart) & JPEG/PNG profile picture \\
  }
  \resheader\quad\textit{Full \texttt{UserProfile} object with updated \texttt{avatar\_url}}
\end{endpoint}

\begin{endpoint}{\PUT}{/users/me/password}{\authrequired}{
  Changes the authenticated user's password after verifying the current one
  against the Firestore bcrypt hash.
}
  \reqheader
  \paramtable{
    current\_password & string & Existing password \\
    new\_password     & string & New password, minimum 8 characters \\
  }
  \resheader
  \paramtable{
    message & string & \texttt{"Password changed successfully"} \\
  }
\end{endpoint}

\begin{endpoint}{\GET}{/users}{\adminonly}{
  Lists all registered users with optional filters. Results are paginated using
  limit/offset and password hashes are stripped from every response.
}
  \reqheader\quad\textit{Query params:} \texttt{role}, \texttt{is\_active}, \texttt{limit} (max 200), \texttt{offset}\\[4pt]
  \resheader
  \paramtable{
    total & int   & Total matching users \\
    items & array & Array of user profile objects \\
  }
\end{endpoint}

\begin{endpoint}{\GET}{/users/\{user\_id\}}{\adminonly}{
  Returns the full profile of any user by their Firestore document ID;
  password hash is stripped from the response.
}
  \reqheader\quad\textit{Path param:} \texttt{user\_id}\\[4pt]
  \resheader\quad\textit{User profile object}
\end{endpoint}

\begin{endpoint}{\PUT}{/users/\{user\_id\}/role}{\adminonly}{
  Reassigns a user to a different role. Use with caution — role changes
  take effect immediately on the next token refresh.
}
  \reqheader
  \paramtable{
    role & enum & \texttt{customer | employee | admin | driver | super\_admin} \\
  }
  \resheader
  \paramtable{
    message & string & Confirmation \\
    role    & string & Newly assigned role \\
  }
\end{endpoint}

\begin{endpoint}{\PATCH}{/users/\{user\_id\}/status}{\adminonly}{
  Activates or deactivates a user account. Deactivated users receive a
  403 on login until reactivated.
}
  \reqheader
  \paramtable{
    is\_active & bool & \texttt{true} to activate, \texttt{false} to deactivate \\
  }
  \resheader
  \paramtable{
    message   & string & Confirmation \\
    is\_active & bool   & New status \\
  }
\end{endpoint}

\begin{endpoint}{\DELETE}{/users/\{user\_id\}}{\adminonly}{
  Permanently deletes a user from Firestore. Admins cannot delete their own
  account; 204 is returned on success with no response body.
}
  \reqheader\quad\textit{Path param:} \texttt{user\_id}\\[4pt]
  \resheader\quad\textit{204 No Content}
\end{endpoint}

\section{Sessions  \normalfont\small\texttt{/api/v1/sessions}}
% ══════════════════════════════════════════════════════════════════════════════

\begin{endpoint}{\POST}{/sessions/generate-qr}{\adminonly}{
  Generates a session QR code for a specific nozzle. The QR encodes a
  signed session ID and has a configurable expiry (default 5 minutes).
}
  \reqheader
  \paramtable{
    nozzle\_id         & string & Target nozzle ID \\
    expires\_in\_seconds & int  & Default 300 \\
  }
  \resheader
  \paramtable{
    session\_id       & string & New session ID \\
    qr\_data          & string & Raw data encoded in the QR \\
    qr\_image\_base64 & string & Base64 PNG of the QR image \\
    expires\_at       & datetime & Expiry timestamp \\
  }
\end{endpoint}

\begin{endpoint}{\POST}{/sessions/scan}{\authrequired}{
  Processes a scanned QR code and activates the corresponding session,
  linking it to the authenticated user or a provided user ID.
}
  \reqheader
  \paramtable{
    qr\_data & string  & Raw QR payload from the scanner \\
    user\_id & string? & Optional override for the linked user \\
  }
  \resheader
  \paramtable{
    session\_id & string & Activated session ID \\
    status      & string & \texttt{"active"} \\
    nozzle\_id  & string & Associated nozzle \\
  }
\end{endpoint}

\begin{endpoint}{\POST}{/sessions/start}{\authrequired}{
  Device-initiated session start — called by the mobile app after connecting
  to the nozzle's WiFi hotspot (SSID: FuelMonitor) via QR nozzle ID.
}
  \reqheader
  \paramtable{
    nozzle\_id & string & Nozzle to start the session on \\
  }
  \resheader
  \paramtable{
    session\_id & string & New session ID \\
    nozzle\_id  & string & Confirmed nozzle ID \\
    status      & string & \texttt{"active"} \\
  }
\end{endpoint}

\begin{endpoint}{\GET}{/sessions/active}{\adminonly}{
  Returns all sessions currently in \texttt{active} status. Used by the admin
  dashboard to monitor live fueling activity across all nozzles.
}
  \reqheader\quad\textit{None}\\[4pt]
  \resheader\quad\textit{Array of session objects}
\end{endpoint}

\begin{endpoint}{\GET}{/sessions/concurrent}{\adminonly}{
  Checks how many concurrent sessions are running on a specific nozzle.
  Used for fraud detection and capacity management.
}
  \reqheader\quad\textit{Query param:} \texttt{nozzle\_id}\\[4pt]
  \resheader
  \paramtable{
    nozzle\_id        & string & Queried nozzle \\
    concurrent\_count & int    & Number of active sessions \\
    sessions          & array  & Session IDs \\
  }
\end{endpoint}

\begin{endpoint}{\GET}{/sessions/\{session\_id\}}{\authrequired}{
  Returns full details of a session including status, timestamps,
  total litres dispensed, total amount, and linked transaction ID.
}
  \reqheader\quad\textit{Path param:} \texttt{session\_id}\\[4pt]
  \resheader
  \paramtable{
    id             & string   & Session ID \\
    nozzle\_id     & string   & Nozzle ID \\
    user\_id       & string   & Linked user \\
    status         & enum     & \texttt{active | completed | failed | pending} \\
    started\_at    & datetime & Start timestamp \\
    ended\_at      & datetime? & End timestamp \\
    total\_litres  & float    & Litres dispensed \\
    total\_amount  & float    & Amount in PKR \\
    transaction\_id & string? & Created transaction ID \\
  }
\end{endpoint}

\begin{endpoint}{\POST}{/sessions/\{session\_id\}/close}{\authrequired}{
  Manually closes an active or pending session. Accepts an optional reason
  string; defaults to \texttt{"manual"}.
}
  \reqheader
  \paramtable{
    reason & string? & Closure reason, default \texttt{"manual"} \\
  }
  \resheader
  \paramtable{
    message & string & Confirmation \\
  }
\end{endpoint}

\begin{endpoint}{\POST}{/sessions/timeout-check}{\noauth}{
  Background job endpoint that auto-closes sessions that have exceeded
  their expiry time without activity. Called by a scheduler.
}
  \reqheader\quad\textit{None}\\[4pt]
  \resheader
  \paramtable{
    timed\_out\_sessions\_closed & int & Count of sessions closed \\
  }
\end{endpoint}

\section{Transactions  \normalfont\small\texttt{/api/v1/transactions}}
% ══════════════════════════════════════════════════════════════════════════════

\begin{endpoint}{\POST}{/transactions}{\authrequired}{
  Records a completed fuel dispensing transaction with litres, price, and
  payment method. Fraud detection runs asynchronously on the created record.
}
  \reqheader
  \paramtable{
    session\_id       & string  & Linked session \\
    nozzle\_id        & string  & Nozzle used \\
    user\_id          & string  & Customer \\
    fuel\_type        & enum    & \texttt{petrol | diesel | cng | lpg} \\
    litres\_dispensed & float   & Must be \textgreater 0 \\
    price\_per\_litre & float   & PKR per litre \\
    payment\_method   & enum    & \texttt{cash | card | wallet | qr\_pay} \\
    vehicle\_id       & string? & Optional fleet vehicle \\
    employee\_id      & string? & Optional serving employee \\
  }
  \resheader
  \paramtable{
    id               & string   & Transaction ID \\
    total\_amount    & float    & \texttt{litres × price} in PKR \\
    status           & enum     & \texttt{completed | pending | failed} \\
    is\_flagged      & bool     & Fraud flag status \\
    created\_at      & datetime & Timestamp \\
  }
\end{endpoint}

\begin{endpoint}{\GET}{/transactions/my}{\authrequired}{
  Returns the authenticated user's own transaction history, paginated.
  Includes completed, pending and failed records.
}
  \reqheader\quad\textit{Query params:} \texttt{limit} (max 100, default 20), \texttt{offset}\\[4pt]
  \resheader
  \paramtable{
    total & int   & Total transactions for this user \\
    items & array & Paginated transaction objects \\
  }
\end{endpoint}

\begin{endpoint}{\GET}{/transactions/\{transaction\_id\}}{\authrequired}{
  Returns the full detail of a single transaction including receipt URL,
  evidence URL, and fraud flag status.
}
  \reqheader\quad\textit{Path param:} \texttt{transaction\_id}\\[4pt]
  \resheader\quad\textit{Full transaction object}
\end{endpoint}

\begin{endpoint}{\GET}{/transactions/\{transaction\_id\}/receipt}{\authrequired}{
  Generates and streams a PDF receipt for the transaction. Receipt includes
  transaction details, station info, and a unique reference number.
}
  \reqheader\quad\textit{Path param:} \texttt{transaction\_id}\\[4pt]
  \resheader\quad\textit{PDF file stream (\texttt{application/pdf})}
\end{endpoint}

\begin{endpoint}{\GET}{/transactions}{\adminonly}{
  Lists all transactions with optional filters. Results include fraud flags
  and are paginated. Password fields are never returned.
}
  \reqheader\quad\textit{Query params:} \texttt{nozzle\_id}, \texttt{user\_id}, \texttt{station\_id}, \texttt{fuel\_type}, \texttt{limit}, \texttt{offset}\\[4pt]
  \resheader
  \paramtable{
    total & int   & Total matching records \\
    items & array & Transaction objects \\
  }
\end{endpoint}

\begin{endpoint}{\POST}{/transactions/\{transaction\_id\}/price-update}{\adminonly}{
  Adjusts the price-per-litre of an existing transaction and recalculates
  the total amount. The reason for the update is stored for audit.
}
  \reqheader
  \paramtable{
    price\_per\_litre & float   & New corrected price \\
    reason            & string? & Audit note \\
  }
  \resheader
  \paramtable{
    message          & string & Confirmation \\
    new\_total\_amount & float & Recalculated total in PKR \\
  }
\end{endpoint}

\section{Stations  \normalfont\small\texttt{/api/v1/stations}}
% ══════════════════════════════════════════════════════════════════════════════

\begin{endpoint}{\GET}{/stations/nearby}{\authrequired}{
  Returns fuel stations within a given radius of the provided coordinates,
  sorted by distance. Optionally filtered by fuel type.
}
  \reqheader\quad\textit{Query params:} \texttt{latitude}, \texttt{longitude}, \texttt{radius\_km} (default 10), \texttt{fuel\_type?}\\[4pt]
  \resheader\quad\textit{Array of station objects with \texttt{distance\_km} field added}
\end{endpoint}

\begin{endpoint}{\GET}{/stations/route}{\authrequired}{
  Finds stations along a driving route by calculating a midpoint buffer
  between origin and destination coordinates.
}
  \reqheader\quad\textit{Query params:} \texttt{origin\_lat}, \texttt{origin\_lng}, \texttt{dest\_lat}, \texttt{dest\_lng}, \texttt{fuel\_type?}\\[4pt]
  \resheader\quad\textit{Array of stations with \texttt{distance\_from\_route\_km}}
\end{endpoint}

\begin{endpoint}{\GET}{/stations/me/favorites}{\authrequired}{
  Returns the authenticated user's list of favourite stations with
  current prices and operating hours.
}
  \reqheader\quad\textit{None}\\[4pt]
